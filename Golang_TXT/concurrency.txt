var wg = sync.WaitGroup

func main() {

    wg.Add(2) // sinalizo que vou add 2 goroutines
    go func1()
    go func2()
    
    // sinalizar que quero que espera as 2 goroutines anunciadas
    // terminarem de executar	
    wg.Wait()

}




package main

import (
	"fmt"
)

func main() {

	messages1 := make(chan string)
	messages2 := make(chan string)

	go func() {
		messages1 <- "ping"
		messages2 <- "deehoo"
	}()

	msg := (<-messages1) + " " + (<-messages2) // block until messages channels are ready and sent the info
	fmt.Println(msg)
}
